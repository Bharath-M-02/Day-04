1. How to compare two JSON have the same properties without order?
a. var obj1 = { name: "Person 1", age:5 };
b. var obj2 = { age:5, name: "Person 1" };

Code:

var flag=true;

if(Object.keys(obj1).length==Object.keys(obj2).length){
    for(key in obj1) {
        if(obj1[key] == obj2[key]) {
            continue;
        }
        else {
            flag=false;
            break;
        }
    }
}
else {
    flag=false;
}


if(flag==true)
    console.log("Both objects are equal");

2. Use the rest countries API url -> https://raw.githubusercontent.com/rvsp/restcountries-json-data/master/res-countries.json and display all the country flags in console

Code:
var xhr = new XMLHttpRequest();
xhr.open(
  "GET",
  "https://raw.githubusercontent.com/rvsp/restcountries-json-data/master/res-countries.json"
);
xhr.send();

xhr.onload = function () {
  if (xhr.status >= 200 && xhr.status < 300) {
    var data = JSON.parse(xhr.response);
    // console.log(data);
    for (let i = 0; i < data.length; i++) {
      var user = data[i];
      console.log(user.flag);
    }
  }
};
